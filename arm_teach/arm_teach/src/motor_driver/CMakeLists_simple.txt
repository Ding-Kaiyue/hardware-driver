cmake_minimum_required(VERSION 3.8)
project(motor_driver)

# Default to C++17
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 17)
endif()

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(geometry_msgs REQUIRED)

# Find system dependencies  
find_package(PkgConfig REQUIRED)
pkg_check_modules(LIBUSB REQUIRED libusb-1.0)
find_package(Threads REQUIRED)

# Include directories
include_directories(
  ${CMAKE_CURRENT_SOURCE_DIR}/../../../include  # Path to damiao headers
  ${LIBUSB_INCLUDE_DIRS}
)

# Build existing motor library
add_library(damiao_motor STATIC
  ${CMAKE_CURRENT_SOURCE_DIR}/../../../src/protocol/damiao.cpp
)

target_include_directories(damiao_motor PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}/../../../include
  ${LIBUSB_INCLUDE_DIRS}
)

target_link_libraries(damiao_motor
  Threads::Threads
  ${CMAKE_CURRENT_SOURCE_DIR}/../../../lib/libu2canfd.a
  ${LIBUSB_LIBRARIES}
)

# Arm controller node (simplified)
add_executable(arm_controller_node
  src/arm_controller_node.cpp
)

target_link_libraries(arm_controller_node
  damiao_motor
)

ament_target_dependencies(arm_controller_node
  rclcpp
  std_msgs
  sensor_msgs
  geometry_msgs
)

# Install targets
install(TARGETS arm_controller_node
  DESTINATION lib/${PROJECT_NAME}
)

# Install launch files
install(DIRECTORY launch
  DESTINATION share/${PROJECT_NAME}
)

# Install config files
install(DIRECTORY config
  DESTINATION share/${PROJECT_NAME}
)

ament_package()
